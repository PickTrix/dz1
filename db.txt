CREATE TABLE users(
	"user_id"	INTEGER NOT NULL UNIQUE,
	"login"	TEXT NOT NULL UNIQUE,
	"password"	TEXT NOT NULL,
	"reg_date"	TEXT NOT NULL,
	"birth_date"	TEXT NOT NULL,
	PRIMARY KEY("user_id")
);

CREATE TABLE "things" (
	"thing_id"	INTEGER NOT NULL UNIQUE,
	"name"	TEXT NOT NULL,
	"amount"	INTEGER NOT NULL,
	"price"	INTEGER NOT NULL,
	"info"	TEXT NOT NULL UNIQUE,
	"discount"	INTEGER NOT NULL,
	"picture_url"	TEXT NOT NULL UNIQUE,
	PRIMARY KEY("thing_id")
);

CREATE TABLE "history" (
	"us_id"	INTEGER NOT NULL UNIQUE,
	"item_id"	INTEGER NOT NULL UNIQUE,
	"kol-vo"	INTEGER NOT NULL,
	"minus_money"	INTEGER NOT NULL,
	"date"	TEXT NOT NULL,
	FOREIGN KEY ("minus_money") REFERENCES things("discount"),
	FOREIGN KEY ("us_id") REFERENCES users("user_id"),
	FOREIGN KEY ("item_id") REFERENCES things("thing_id")
);
